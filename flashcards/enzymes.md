    concept_id: c-substrate-binding

    tags: [binding, models]
  - id: bch-020
    concept_id: c-transition-state
    front: "How do enzymes catalyze reactions?"

    concept_id: c-michaelis-menten
    front: "What does Km represent in enzyme kinetics?"

    tags: [kinetics, Km]
  - id: bch-022
    concept_id: c-lineweaver-burk
    front: "What does a Lineweaver-Burk plot show?"

    tags: [kinetics, plot]
  - id: bch-023
    concept_id: c-competitive-inhibition
    front: "How does competitive inhibition affect Km and Vmax?"
    back: "Km increases (apparent), Vmax unchanged."
    tags: [inhibition, competitive]
  - id: bch-024
    concept_id: c-noncompetitive-inhibition
    front: "How does non-competitive inhibition affect Km and Vmax?"
    back: "Km unchanged, Vmax decreases."
    tags: [inhibition, non-competitive]
  - id: bch-025
    concept_id: c-uncompetitive-inhibition
    front: "How does uncompetitive inhibition affect Km and Vmax?"
    back: "Both Km and Vmax decrease."
    tags: [inhibition, uncompetitive]
  - id: bch-026
    concept_id: c-cofactors
    front: "What is the difference between a coenzyme and a prosthetic group?"
    back: "Coenzymes bind and release during catalysis; prosthetic groups are covalently attached."
    tags: [cofactors, types]
  - id: bch-027
    concept_id: c-enzyme-classification
    front: "How many EC classes are there?"
    back: "Seven (Oxidoreductases, Transferases, Hydrolases, Lyases, Isomerases, Ligases, Translocases)."
    tags: [classification, EC]
  - id: bch-028
    concept_id: c-enzyme-catalysis
